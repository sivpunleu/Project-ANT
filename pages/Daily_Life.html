ការអនុវត្តព្រះធម៌ក្នុងជីវិតប្រចាំថ្ងៃ

សិល្បៈនៃការរស់នៅប្រកបដោយសតិ និងបញ្ញា

ត្រឡប់ក្រោយ

ការយកព្រះធម៌មកប្រើប្រាស់ក្នុងជីវិតប្រចាំថ្ងៃ គឺការអនុវត្តគោលការណ៍សំខាន់ៗ ដូចជា សីល (ការប្រព្រឹត្តល្អ), សមាធិ (ការតម្កល់ចិត្ត) និង បញ្ញា (ការយល់ដឹង) ទៅក្នុងគ្រប់សកម្មភាពរបស់យើង។
ព្រះធម៌មិនមែនសម្រាប់តែនៅក្នុងវត្ត ឬពេលសូត្រមន្តទេ តែគឺសម្រាប់អនុវត្តគ្រប់ជំហាននៃជីវិត ដើម្បីបង្កើតសេចក្ដីសុខ និងសន្តិភាព។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

វិធីសាស្ត្ររៀនពូកែតាមបែបព្រះពុទ្ធសាសនា៖
១. សតិ (Mindfulness)

ផ្ចង់អារម្មណ៍ពេលកំពុងរៀន ជៀសវាងការរវើរវាយ ឬលេងទូរស័ព្ទ។

២. វិរិយៈ (Effort)

មានសេចក្ដីព្យាយាម ឧស្សាហ៍រៀនសូត្រ មិនបាក់ទឹកចិត្តពេលជួបមេរៀនពិបាក។

៣. សម្មាទិដ្ឋិ (Right View)

យល់ដឹងថាការសិក្សាគឺដើម្បីចម្រើនបញ្ញា មិនមែនត្រឹមតែប្រឡងយកពិន្ទុ។

ដំបូន្មាន៖ សិក្សាដើម្បី "បញ្ញា" (Wisdom) យកទៅជួយសង្គម មិនមែនត្រឹមតែដើម្បី "ចំណេះ" (Knowledge) ទេ។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

ការធ្វើការងារឲ្យមានប្រសិទ្ធភាព និងសប្បាយចិត្ត៖
១. សម្មាអាជីវោ (Right Livelihood)

ប្រកបការងារដោយស្មោះត្រង់ ទៀងត្រង់ មិនបោកប្រាស់ ឬកេងបន្លំ។

២. ខន្តី (Patience)

មានការអត់ធ្មត់ខ្ពស់ ពេលជួបបញ្ហាក្នុងការងារ ឬជាមួយអតិថិជន។

៣. មេត្តា (Loving-kindness)

ចេះជួយយកអាសាគ្នាទៅវិញទៅមក បង្កើតបរិយាកាសល្អក្នុងកន្លែងធ្វើការ។

ដំបូន្មាន៖ ធ្វើការងារដើម្បីផ្ដល់តម្លៃ និងជួយសង្គម មិនមែនត្រឹមតែដើម្បីប្រាក់ខែទេ។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

កសាងសុភមង្គលក្នុងគ្រួសារតាមផ្លូវធម៌៖
១. កតញ្ញូ (Gratitude)

គោរព ដឹងគុណ និងជួយមើលថែឪពុកម្ដាយ ឬចាស់ទុំក្នុងគ្រួសារ។

២. សង្គហធម៌ (Helpfulness)

ចេះជួយទុក្ខធុរៈគ្នាទៅវិញទៅមក ចែករំលែក និងយោគយល់អធ្យាស្រ័យ។

៣. សច្ចៈ (Truthfulness)

និយាយពាក្យពិត មិនកុហកបោកប្រាស់សមាជិកគ្រួសារ បង្កើតទំនុកចិត្ត។

ដំបូន្មាន៖ ផ្ទះដែលសាងលើព្រះធម៌ គឺជាផ្ទះដែលពោរពេញដោយសន្តិភាព និងសុភមង្គល។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

ការរក្សាសីល ៥ ឱ្យបានបរិសុទ្ធ៖
១. មិនសម្លាប់: មានចិត្តមេត្តា មិនបៀតបៀនសត្វ។
២. មិនលួច: មានភាពស្មោះត្រង់ មិនយករបស់គេ។
៣. មិនប្រព្រឹត្តខុសក្នុងកាម: ស្មោះត្រង់នឹងដៃគូ។
៤. មិននិយាយកុហក: និយាយតែពាក្យពិត និងមានប្រយោជន៍។
៥. មិនសេពគ្រឿងស្រវឹង: វៀរចាកគ្រឿងញៀន ដែលធ្វើឲ្យបាត់បង់សតិ។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

ការធ្វើទានមិនចាំបាច់ទាល់តែមានលុយទេ៖
  • អាមិសទាន: ការបរិច្ចាគទ្រព្យ អាហារ ដល់អ្នកខ្វះខាត។
  • ធម្មទាន: ការចែករំលែកចំណេះដឹង ឬដំបូន្មានល្អៗ។
  • អភ័យទាន: ការផ្ដល់ការអភ័យទោស មិនចងគំនុំ។
ដំបូន្មាន៖ ធ្វើទានដោយចិត្តជ្រះថ្លា ដោយមិនរំពឹងការតបស្នង នោះទើបជាបុណ្យដ៏ធំ។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

គន្លឹះក្នុងការហ្វឹកហាត់ចិត្ត៖
ពេលព្រឹក

ចំណាយពេល ៥-១០ នាទី អង្គុយស្ងៀម ផ្ចង់អារម្មណ៍លើដង្ហើមចេញចូល។

ពេលថ្ងៃ

មានសតិដឹងខ្លួន ពេលដើរ ពេលញ៉ាំអាហារ (ដឹងរសជាតិ) និងពេលធ្វើការ។

ពេលល្ងាច

រំឭកពីទង្វើក្នុងថ្ងៃ អភ័យទោសដល់ខ្លួនឯងនិងអ្នកដទៃ រួចបណ្ដុះមេត្តាចិត្ត។

តើចម្លើយនេះមានប្រយោជន៍ទេ?

ដំណោះស្រាយបញ្ហាតាមអរិយសច្ច ៤៖
១. ទទួលស្គាល់ (ទុក្ខ): ហ៊ានទទួលស្គាល់ការពិត ថាបញ្ហាបានកើតឡើងហើយ។
២. រកមូលហេតុ (សមុទ័យ): ប្រើបញ្ញាពិចារណារកដើមហេតុ ដោយមិនបន្ទោសអ្នកដទៃ។
៣. ដោះស្រាយ (មគ្គ): ប្រើសតិ និងសេចក្តីព្យាយាម ដោះស្រាយមួយជំហានម្តងៗ។
ដំបូន្មាន៖ យើងមិនអាចគ្រប់គ្រងអ្វីដែលកើតឡើងបានទេ តែយើងអាចគ្រប់គ្រង "ប្រតិកម្ម" របស់យើងចំពោះវាបាន។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

វិធីរម្ងាប់កំហឹង៖
  • សតិ (Mindfulness): ដឹងឲ្យទាន់ថា "ខ្ញុំកំពុងខឹង"។ គ្រាន់តែ "ដឹង" កុំទាន់ "តបត"។
  • មេត្តា (Metta): ពិចារណាថា គេធ្វើដូច្នេះព្រោះគេមានទុក្ខ ឬខ្វះការយល់ដឹង។
  • ឧបេក្ខា (Equanimity): យល់ថា យើងមិនអាចផ្លាស់ប្ដូរគេបានទេ តែយើងអាចរក្សាចិត្តយើងកុំឲ្យក្ដៅក្រហាយបាន។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

ប្រយោជន៍ដ៏ធំបំផុតគឺ៖
  • សន្តិភាពក្នុងចិត្ត៖ បន្ថយភាពតានតឹង (Stress) និងកំហឹង។
  • ទំនាក់ទំនងល្អ៖ ធ្វើឲ្យគ្រួសារ និងកន្លែងធ្វើការ កាន់តែចុះសម្រុងគ្នា។
  • បញ្ញាចម្រើន៖ ធ្វើការសម្រេចចិត្តបានត្រឹមត្រូវជាងមុន។
  • ជីវិតមានន័យ៖ រស់នៅដោយមានគោលបំណង និងក្ដីសុខពិតប្រាកដ។
សរុបមក៖ ការអនុវត្តព្រះធម៌ គឺ "សិល្បៈនៃការរស់នៅ" ដែលនាំមកនូវសេចក្ដីសុខ ទាំងបច្ចុប្បន្ន និងអនាគត។
តើចម្លើយនេះមានប្រយោជន៍ទេ?

មិនមានលទ្ធផលសម្រាប់ការស្វែងរកនេះទេ។

អនុវត្តធម៌ក្នុងគ្រប់កាលៈទេសៈ

ក្នុងការសិក្សា

ប្រើ "សតិ" ដើម្បីផ្ចង់អារម្មណ៍ និង "វិរិយៈ" ដើម្បីមានការព្យាយាមក្នុងការរៀនសូត្រ។

ក្នុងការងារ

ប្រើ "សម្មាអាជីវោ" (ការងារស្មោះត្រង់) និង "ខន្តី" (ការអត់ធ្មត់) ក្នុងកន្លែងធ្វើការ។

ក្នុងគ្រួសារ

ប្រើ "មេត្តា" (សេចក្ដីស្រឡាញ់) និង "កតញ្ញូ" (ការដឹងគុណ) ដើម្បីសុភមង្គលក្នុងផ្ទះ។

ត្រៀមខ្លួនចាប់ផ្ដើមអនុវត្តឬនៅ?

ព្រះធម៌ពិត គឺកើតឡើងពីការអនុវត្ត។ សូមចាប់ផ្ដើមពីចំណុចតូចៗ ដូចជាការរក្សាសីល ឬការចម្រើនសតិ ៥នាទី ក្នុងមួយថ្ងៃ។
